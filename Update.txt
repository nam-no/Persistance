function menu{
    Write-Host "What do you want to do ?"
    Write-Host "1. Update Action ."
    Write-Host "2. Update Trigger"
}
function Update_TaskSchedul_Action{
    param(
        [string]$TaskName,
        [string]$ActionEx,
        [string]$Argument = " "
    )

    $action = New-ScheduledTaskAction -Execute $ActionEx -Argument $Argument

    Set-ScheduledTask -TaskName $TaskName -Action $action 
    
}
function Update_TaskSchedul_Trigger{
    param(
        [string]$TaskName,
        [string]$TriggerTime
    )
    $trigger = New-ScheduledTaskTrigger -Daily -At $TriggerTime
    Set-ScheduledTask -TaskName $TaskName -Trigger $trigger 
    
}

if($choose -eq 1){
    [string]$taskName = Read-Host -Prompt "Enter the task name"
    [string]$action = Read-Host -Prompt "Enter the action (excutable path)"
    [string]$argument = Read-Host -Prompt "Enter the Argument (optional)"
    Update_TaskSchedul_Action -TaskName $taskName -ActionEx $action -Argument $argument
}elseif($choose -eq 2){
    [string]$taskName = Read-Host -Prompt "Enter the task name"
    [string]$triggertime = Read-Host -Prompt "Enter the Trigger Time (e.g., 12:00PM)"
    Update_TaskSchedul_Trigger -TaskName $taskName -TriggerTime $triggertime
}else {return}
